``` react
import { Grid } from '@mui/material';

import { useEffect, useState } from 'react';
import { Outlet } from 'react-router-dom';

import Category from '@layouts/Category/Category';
import Login from '@layouts/Login/Login';
import Copyright from '@layouts/Copyright/Copyright';

import Loading from '@components/Loading/Loading';

import { API } from '@api/api';

const Main = () => {
	const [loading, setLoading] = useState<boolean>(false);

	useEffect(() => {
		window.scrollTo(0, 0);
		API.interceptors.request.use(
			(config) => {
				// 팝업(모달)창 띄우는 경우 로딩창 띄우지 않음
				if (config.data.mode === 'detail') {
					setLoading(false);
				} else {
					setLoading(true);
				}

				return config;
			},
			(error) => {
				setLoading(false);
				return Promise.reject(error);
			}
		);

		API.interceptors.response.use(
			(config) => {
				return config;
			},
			(error) => {
				setLoading(false);
				return Promise.reject(error);
			}
		);

		setTimeout(() => {
			setLoading(false);
		}, 1500);
	}, [loading]);

	return (
		<>
			<Loading loading={loading} />

			<Grid container spacing={1}>
				<Grid item xs={2}>
					<Login></Login>
					<Category></Category>
				</Grid>

				<Grid item xs={10}>
					<Outlet />
				</Grid>
			</Grid>

			<Grid container>
				<Grid item xs={12}>
					<Copyright></Copyright>
				</Grid>
			</Grid>
		</>
	);
};

export default Main;

```
